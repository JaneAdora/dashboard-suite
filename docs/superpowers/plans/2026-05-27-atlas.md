//! Top-level app: AppState, event loop, view dispatch.
use anyhow::Result;
use crossterm::event::{self, Event, KeyCode, KeyEventKind, KeyModifiers};
use ratatui::text::Line;
use ratatui::widgets::Paragraph;
use std::time::Duration;

pub struct App {
    pub should_quit: bool,
}

impl App {
    pub fn new() -> Self {
        Self { should_quit: false }
    }
}

pub fn run<B: ratatui::backend::Backend>(
    terminal: &mut ratatui::Terminal<B>,
    app: &mut App,
) -> Result<()> {
    while !app.should_quit {
        terminal.draw(|f| {
            f.render_widget(
                Paragraph::new(Line::from("atlas (press q to quit)")),
                f.area(),
            );
        })?;

        if event::poll(Duration::from_millis(100))? {
            if let Event::Key(key) = event::read()? {
                if key.kind == KeyEventKind::Release {
                    continue;
                }
                if key.code == KeyCode::Char('q')
                    || (key.modifiers.contains(KeyModifiers::CONTROL)
                        && key.code == KeyCode::Char('c'))
                {
                    app.should_quit = true;
                }
            }
        }
    }
    Ok(())
}
```

use crossterm::terminal::{
    disable_raw_mode, enable_raw_mode, EnterAlternateScreen, LeaveAlternateScreen,
};
use std::io::stdout;

mod app;
mod theme;

fn main() -> Result<()> {
    // Panic hook restores terminal so a panic doesn't leave the user stuck in raw mode.
    let default_hook = std::panic::take_hook();
    std::panic::set_hook(Box::new(move |info| {
        let _ = disable_raw_mode();
        let _ = execute!(stdout(), LeaveAlternateScreen);
        default_hook(info);
    }));

    enable_raw_mode()?;
    let mut out = stdout();
    execute!(out, EnterAlternateScreen, crossterm::terminal::SetTitle("atlas"))?;
    let backend = ratatui::backend::CrosstermBackend::new(out);
    let mut terminal = ratatui::Terminal::new(backend)?;

    let mut app = app::App::new();
    let result = app::run(&mut terminal, &mut app);

    disable_raw_mode()?;
    execute!(stdout(), LeaveAlternateScreen)?;
    result
}

//! Frontmatter + markdown parsing.
//!
//! Frontmatter is hand-rolled (the schema is 4 fixed fields; serde_yaml is
//! deprecated upstream and a full YAML parser is overkill).

use crate::library::doc::{DocType, Status};

#[derive(Debug, Clone, Default, PartialEq)]
pub struct Frontmatter {
    pub doc_type: Option<DocType>,
    pub title: Option<String>,
    pub status: Option<Status>,
    pub progress: Option<u8>,
}

/// Split a markdown source into (frontmatter, body) by recognizing a leading
/// `---\n...---\n` block.  Returns `(None, full_source)` if no frontmatter.
pub fn split_frontmatter(src: &str) -> (Option<&str>, &str) {
    let rest = match src.strip_prefix("---\n") {
        Some(r) => r,
        None => return (None, src),
    };
    // Find the closing "---\n" or "---\r\n" at line start.
    let mut search_from = 0;
    while let Some(idx) = rest[search_from..].find("---") {
        let abs = search_from + idx;
        let at_line_start = abs == 0 || rest.as_bytes()[abs - 1] == b'\n';
        let after = &rest[abs + 3..];
        let valid_end = after.starts_with('\n') || after.starts_with("\r\n") || after.is_empty();
        if at_line_start && valid_end {
            let fm = &rest[..abs];
            let body_start = abs
                + 3
                + if after.starts_with("\r\n") { 2 } else if after.starts_with('\n') { 1 } else { 0 };
            return (Some(fm), &rest[body_start..]);
        }
        search_from = abs + 3;
    }
    (None, src)
}

/// Parse a frontmatter block into a Frontmatter struct.  Unknown keys are
/// ignored; values are trimmed; quoted strings have their quotes stripped.
pub fn parse_frontmatter(block: &str) -> Frontmatter {
    let mut fm = Frontmatter::default();
    for line in block.lines() {
        let line = line.trim();
        if line.is_empty() || line.starts_with('#') {
            continue;
        }
        let (key, value) = match line.split_once(':') {
            Some(kv) => kv,
            None => continue,
        };
        let key = key.trim();
        let value = value.trim().trim_matches(|c| c == '"' || c == '\'');
        match key {
            "type" => {
                fm.doc_type = match value {
                    "plan" => Some(DocType::Plan),
                    "roadmap" => Some(DocType::Roadmap),
                    "task" => Some(DocType::Task),
                    _ => None,
                };
            }
            "title" => fm.title = Some(value.to_string()),
            "status" => {
                fm.status = match value {
                    "active" => Some(Status::Active),
                    "shipped" => Some(Status::Shipped),
                    "dropped" => Some(Status::Dropped),
                    _ => None,
                };
            }
            "progress" => fm.progress = value.parse::<u8>().ok().map(|p| p.min(100)),
            _ => {}
        }
    }
    fm
}

use crate::library::doc::{Heading, TaskLine};

/// Walk a markdown body, returning headings and checkbox task lines.
/// Skips checkbox-like lines inside fenced code blocks.
pub fn parse_body(body: &str) -> (Vec<Heading>, Vec<TaskLine>) {
    let mut headings = Vec::new();
    let mut tasks = Vec::new();
    let mut in_code_fence = false;

    for (i, raw_line) in body.lines().enumerate() {
        let line = raw_line.trim_end();

        // Track fenced code blocks.
        let stripped = line.trim_start();
        if stripped.starts_with("```") || stripped.starts_with("~~~") {
            in_code_fence = !in_code_fence;
            continue;
        }
        if in_code_fence {
            continue;
        }

        // ATX heading: 1-6 '#' followed by space.
        if let Some(rest) = strip_heading_prefix(line) {
            let level = line.len() - rest.len() - 1; // '#' count
            headings.push(Heading {
                level: level as u8,
                text: rest.trim_end_matches('#').trim().to_string(),
                line: i,
            });
            continue;
        }

        // Checkbox task: ws + (- or *) + ws + [ x|X|space ] + ws + text.
        if let Some((checked, text)) = parse_checkbox(line) {
            tasks.push(TaskLine { line: i, checked, text });
        }
    }

    (headings, tasks)
}

fn strip_heading_prefix(line: &str) -> Option<&str> {
    let trimmed = line.trim_start();
    let hashes = trimmed.bytes().take_while(|&b| b == b'#').count();
    if hashes == 0 || hashes > 6 {
        return None;
    }
    let rest = &trimmed[hashes..];
    if rest.starts_with(' ') {
        Some(&rest[1..])
    } else {
        None
    }
}

fn parse_checkbox(line: &str) -> Option<(bool, String)> {
    let s = line.trim_start();
    let s = s.strip_prefix("- ").or_else(|| s.strip_prefix("* "))?;
    let s = s.strip_prefix('[')?;
    let (state_ch, rest) = {
        let mut chars = s.chars();
        let c = chars.next()?;
        (c, chars.as_str())
    };
    let rest = rest.strip_prefix("] ")?;
    let checked = matches!(state_ch, 'x' | 'X');
    let is_space = state_ch == ' ';
    if !(checked || is_space) {
        return None;
    }
    Some((checked, rest.to_string()))
}
```

//! Progress percentage rule: checkbox-wins-over-frontmatter.
use crate::library::doc::TaskLine;

/// Compute progress percentage.  If any tasks exist, derive from checkboxes.
/// Otherwise fall back to the frontmatter `progress` value.  Returns None if
/// neither is available; the index then shows a dim em-equivalent in place
/// of the bar.
pub fn compute(tasks: &[TaskLine], frontmatter_progress: Option<u8>) -> Option<u8> {
    if !tasks.is_empty() {
        let done = tasks.iter().filter(|t| t.checked).count();
        let pct = (done as f32 / tasks.len() as f32 * 100.0).round() as u8;
        return Some(pct.min(100));
    }

impl Doc {
    /// Parse a markdown source into a Doc.  Returns None if the frontmatter
    /// gate fails (no `---` block or `type:` is missing/invalid).
    pub fn from_str(path: std::path::PathBuf, mtime: std::time::SystemTime, src: &str) -> Option<Doc> {
        use crate::library::parse::{parse_body, parse_frontmatter, split_frontmatter};
        use crate::library::progress;

        let (fm_block, body) = split_frontmatter(src);
        let fm_block = fm_block?;
        let fm = parse_frontmatter(fm_block);
        let doc_type = fm.doc_type?;

        let title = fm
            .title
            .or_else(|| {
                path.file_stem()
                    .and_then(|s| s.to_str())
                    .map(|s| s.to_string())
            })
            .unwrap_or_else(|| "(untitled)".to_string());

        let status = fm.status.unwrap_or_default();
        let (headings, tasks) = parse_body(body);
        let progress_pct = progress::compute(&tasks, fm.progress);

        let rel_label = build_rel_label(&path);

        Some(Doc {
            path,
            rel_label,
            doc_type,
            title,
            status,
            progress: progress_pct,
            headings,
            tasks,
            body: body.to_string(),
            mtime,
        })
    }
}

    let stem = path
        .file_stem()
        .and_then(|s| s.to_str())
        .unwrap_or("(unknown)");

    if let Some(home) = dirs::home_dir() {

//! Walk glob roots, sniff frontmatter, return parsed Doc list.
use crate::library::Doc;
use anyhow::Result;
use std::path::PathBuf;

/// Walk each glob root, parse matching files into Docs, drop those without a
/// valid `type:` frontmatter (Doc::from_str returns None).
pub fn discover(roots: &[String]) -> Result<Vec<Doc>> {
    let mut docs = Vec::new();
    let mut seen = std::collections::HashSet::new();

    for pattern in roots {
        let paths = match glob::glob(pattern) {
            Ok(p) => p,
            Err(_) => continue,
        };
        for entry in paths {
            let path = match entry {
                Ok(p) => p,
                Err(_) => continue,
            };
            if !seen.insert(path.clone()) {
                continue;
            }
            if let Some(doc) = read_and_parse(path) {
                docs.push(doc);
            }
        }
    }
    Ok(docs)
}

fn read_and_parse(path: PathBuf) -> Option<Doc> {
    let meta = std::fs::metadata(&path).ok()?;
    if !meta.is_file() {
        return None;
    }
    let mtime = meta.modified().ok()?;
    let src = std::fs::read_to_string(&path).ok()?;
    // Cheap reject: must start with "---\n".
    if !src.starts_with("---\n") && !src.starts_with("---\r\n") {
        return None;
    }
    Doc::from_str(path, mtime, &src)
}

    }

//! Right-aligned 5-segment bar + fixed-width 4-char percentage.
//!
//! Renders as e.g. "▰▰▰▰▱  80%" (length 11 chars: 5 bar + 2 gap + 4 pct).
//! Pct is always 4 chars wide (right-aligned) so bar end and pct end land
//! on the same column regardless of digit count.
use crate::theme;
use ratatui::style::Style;
use ratatui::text::{Line, Span};

pub const BAR_WIDTH: usize = 5;
pub const FORMATTED_WIDTH: usize = BAR_WIDTH + 2 + 4; // bar + 2 spaces + pct

/// Render a progress value as a styled Line.  `None` renders as a placeholder
/// dim dash spanning FORMATTED_WIDTH.
pub fn render(value: Option<u8>) -> Line<'static> {
    let pct = match value {
        Some(p) => p,
        None => {
            let dash = format!("{:>w$}", "-", w = FORMATTED_WIDTH);
            return Line::from(Span::styled(dash, Style::default().fg(theme::DIM)));
        }
    };

    // Map 0..=100 to 0..=5 filled segments, banker's-ish rounding via .round().
    let filled = ((pct as f32 / 100.0) * BAR_WIDTH as f32).round() as usize;
    let filled = filled.min(BAR_WIDTH);
    let bar: String = std::iter::repeat('▰').take(filled)
        .chain(std::iter::repeat('▱').take(BAR_WIDTH - filled))
        .collect();
    let pct_str = format!("{:>4}", format!("{}%", pct));

    Line::from(vec![
        Span::styled(bar, Style::default().fg(theme::CHECK)),
        Span::raw("  "),
        Span::styled(pct_str, Style::default().fg(theme::CHECK)),
    ])
}

use ratatui::text::{Line, Span};
use ratatui::widgets::{Block, Borders, Paragraph};
use ratatui::Frame;

const NAME_COL_WIDTH: usize = 18;

#[derive(Debug, Clone, Default)]
pub struct IndexState {
    /// Index into the flat ordered list returned by `flat_focus_list`.
    pub focus: usize,
    pub filter: Option<String>,
}

pub fn render(f: &mut Frame, area: Rect, lib: &Library, state: &IndexState) {
    let block = Block::default()
        .borders(Borders::ALL)
        .border_style(Style::default().fg(theme::DIM))
        .title(Span::styled(
            " atlas · library ",
            Style::default().fg(theme::TITLE).add_modifier(theme::TITLE_MOD),
        ));
    let inner = block.inner(area);
    f.render_widget(block, area);

    let chunks = Layout::vertical([
        Constraint::Min(1),
        Constraint::Length(1),
        Constraint::Length(1),
    ])
    .split(inner);

    let body = chunks[0];
    let counts_row = chunks[1];
    let footer_row = chunks[2];

    let mut lines: Vec<Line> = Vec::new();
    let flat = flat_focus_list(lib, state.filter.as_deref());

    let mut flat_i = 0;
    for (label, section) in [
        ("PLANS", lib.plans()),
        ("ROADMAPS", lib.roadmaps()),
        ("TASKS", lib.tasks()),
    ] {
        lines.push(Line::from(""));
        lines.push(Line::from(vec![
            Span::styled("▎ ", Style::default().fg(theme::ROSE)),
            Span::styled(label, Style::default().fg(theme::ROSE).add_modifier(Modifier::BOLD)),
        ]));
        lines.push(Line::from(""));
        for doc in section {
            let visible = state
                .filter
                .as_deref()
                .map(|f| matches_filter(doc, f))
                .unwrap_or(true);
            if !visible {
                continue;
            }
            let focused = flat.get(state.focus).map(|p| *p == doc.path) == Some(true);
            lines.push(row_line(doc, focused));
            let _ = flat_i; // bookkeeping placeholder; flat_focus_list keeps order in sync
            flat_i += 1;
        }
    }

    let para = Paragraph::new(lines).style(Style::default().fg(theme::TEXT).bg(theme::BG_BASE));
    f.render_widget(para, body);

    let counts = format!(" {} docs indexed", lib.docs.len());
    f.render_widget(
        Paragraph::new(counts).style(Style::default().fg(theme::DIM).bg(theme::BG_BASE)),
        counts_row,
    );

    let footer = match &state.filter {
        Some(text) => format!(" /{}_", text),
        None => " j/k move · enter open · / find · q quit".to_string(),
    };
    f.render_widget(
        Paragraph::new(footer).style(Style::default().fg(theme::DIM).bg(theme::BG_BASE)),
        footer_row,
    );
}

fn row_line<'a>(doc: &Doc, focused: bool) -> Line<'a> {
    let marker = if focused { "▸ " } else { "  " };
    let mut name = doc.title.clone();
    if name.chars().count() > NAME_COL_WIDTH {
        name = name.chars().take(NAME_COL_WIDTH - 1).collect::<String>() + "…";
    } else {
        let pad = NAME_COL_WIDTH - name.chars().count();
        name.push_str(&" ".repeat(pad));
    }

    let bar = progress_bar::render(doc.progress);

    let mut spans = vec![
        Span::raw(" "),
        Span::raw(marker.to_string()),
        Span::styled(
            name,
            Style::default().fg(if doc.status == crate::library::Status::Active {
                theme::LAV
            } else {
                theme::DIM
            }),
        ),
        Span::raw(" "),
    ];
    spans.extend(bar.spans);

    let mut line = Line::from(spans);
    if focused {
        line = line.style(Style::default().bg(theme::BG_FOCUS));
    }
    line
}

fn matches_filter(doc: &Doc, q: &str) -> bool {
    let needle = q.to_lowercase();
    doc.title.to_lowercase().contains(&needle) || doc.rel_label.to_lowercase().contains(&needle)
}

/// The flat focus order matches what's rendered: PLANS section, ROADMAPS, TASKS.
pub fn flat_focus_list(lib: &Library, filter: Option<&str>) -> Vec<std::path::PathBuf> {
    let mut out = Vec::new();
    for section in [lib.plans(), lib.roadmaps(), lib.tasks()] {
        for d in section {
            if filter.map(|f| matches_filter(d, f)).unwrap_or(true) {
                out.push(d.path.clone());
            }
        }
    }
    out
}

pub fn run<B: ratatui::backend::Backend>(
    terminal: &mut ratatui::Terminal<B>,
    app: &mut App,
) -> Result<()> {
    while !app.should_quit {
        terminal.draw(|f| {
            // Paint a base-color block so cells we don't render still get the theme bg.
            f.render_widget(Block::default().style(Style::default().bg(theme::BG_BASE)), f.area());
            match &app.view {
                View::Index => index::render(f, f.area(), &app.library, &app.index_state),
                View::Reader { .. } => {
                    let placeholder = Paragraph::new("(reader: implemented in a later task)");
                    f.render_widget(placeholder, f.area());
                }
            }
        })?;

        if event::poll(Duration::from_millis(100))? {
            if let Event::Key(key) = event::read()? {
                if key.kind == KeyEventKind::Release {
                    continue;
                }
                if key.modifiers.contains(KeyModifiers::CONTROL) && key.code == KeyCode::Char('c') {
                    app.should_quit = true;
                    continue;
                }
                handle_key(app, key.code);
            }
        }
    }
    Ok(())
}

fn handle_key(app: &mut App, code: KeyCode) {
    match &app.view {
        View::Index => handle_index_key(app, code),
        View::Reader { .. } => handle_reader_key(app, code),
    }
}

fn handle_index_key(app: &mut App, code: KeyCode) {
    let flat = index::flat_focus_list(&app.library, app.index_state.filter.as_deref());
    let total = flat.len();
    match code {
        KeyCode::Char('q') => app.should_quit = true,
        KeyCode::Char('j') | KeyCode::Down => {
            if total > 0 {
                app.index_state.focus = (app.index_state.focus + 1) % total;
            }
        }
        KeyCode::Char('k') | KeyCode::Up => {
            if total > 0 {
                app.index_state.focus = (app.index_state.focus + total - 1) % total;
            }
        }
        KeyCode::Char('g') => app.index_state.focus = 0,
        KeyCode::Char('G') => {
            if total > 0 {
                app.index_state.focus = total - 1;
            }
        }
        KeyCode::Enter | KeyCode::Char('l') | KeyCode::Right => {
            if let Some(p) = app.focused_path() {
                app.view = View::Reader { path: p.to_path_buf() };
            }
        }
        _ => {}
    }
}

fn handle_reader_key(app: &mut App, code: KeyCode) {

        KeyCode::Char('h')
        | KeyCode::Left
        | KeyCode::Esc
        | KeyCode::Backspace => {
            app.view = View::Index;
        }
        _ => {}
    }

//! Reader view: render a single Doc with section gutters, indented prose,
//! and inline checkboxes.
use crate::library::Doc;
use crate::theme;
use ratatui::layout::{Constraint, Layout, Rect};
use ratatui::style::{Modifier, Style};
use ratatui::text::{Line, Span};
use ratatui::widgets::{Block, Borders, Paragraph};
use ratatui::Frame;

#[derive(Debug, Default, Clone)]
pub struct ReaderState {
    pub scroll: u16,        // top line offset
    pub focused_task: Option<usize>,
}

pub fn render(f: &mut Frame, area: Rect, doc: &Doc, state: &ReaderState) {
    let block = Block::default()
        .borders(Borders::ALL)
        .border_style(Style::default().fg(theme::DIM))
        .title(Span::styled(
            format!(" atlas · {} ", file_label(doc)),
            Style::default().fg(theme::TITLE).add_modifier(theme::TITLE_MOD),
        ));
    let inner = block.inner(area);
    f.render_widget(block, area);

    let chunks = Layout::vertical([
        Constraint::Length(2),
        Constraint::Length(1),
        Constraint::Min(1),
        Constraint::Length(1),
        Constraint::Length(1),
    ])
    .split(inner);

    let header = chunks[0];
    let divider = chunks[1];
    let body = chunks[2];
    let div2 = chunks[3];
    let footer = chunks[4];

    // Header band: title (line 1), meta (line 2).
    let title_line = Line::from(Span::styled(
        format!(" {}", doc.title),
        Style::default().fg(theme::TITLE).add_modifier(theme::TITLE_MOD),
    ));
    let pct = doc.progress.map(|p| format!(" · {p}% done")).unwrap_or_default();
    let meta_line = Line::from(Span::styled(
        format!(
            " {} · {} lines{}",
            doc_type_label(doc),
            doc.body.lines().count(),
            pct
        ),
        Style::default().fg(theme::DIM),
    ));
    f.render_widget(
        Paragraph::new(vec![title_line, meta_line]).style(Style::default().bg(theme::BG_BASE)),
        header,
    );

    let divider_str: String = "─".repeat(divider.width as usize);
    f.render_widget(
        Paragraph::new(divider_str.clone()).style(Style::default().fg(theme::DIM).bg(theme::BG_BASE)),
        divider,
    );

    // Body lines.
    let lines = render_body_lines(doc, state);
    let para = Paragraph::new(lines)
        .style(Style::default().fg(theme::TEXT).bg(theme::BG_BASE))
        .scroll((state.scroll, 0));
    f.render_widget(para, body);

    f.render_widget(
        Paragraph::new(divider_str).style(Style::default().fg(theme::DIM).bg(theme::BG_BASE)),
        div2,
    );
    f.render_widget(
        Paragraph::new(" j/k · space · e · o outline · h ←").style(Style::default().fg(theme::DIM).bg(theme::BG_BASE)),
        footer,
    );
}

fn file_label(doc: &Doc) -> String {
    doc.path
        .file_name()
        .and_then(|n| n.to_str())
        .unwrap_or("(unknown)")
        .to_string()
}

fn doc_type_label(doc: &Doc) -> &'static str {
    match doc.doc_type {
        crate::library::DocType::Plan => "plan",
        crate::library::DocType::Roadmap => "roadmap",
        crate::library::DocType::Task => "task",
    }
}

pub fn render_body_lines<'a>(doc: &'a Doc, state: &ReaderState) -> Vec<Line<'a>> {
    let mut out = Vec::new();
    for (i, raw) in doc.body.lines().enumerate() {
        out.push(line_for(raw, i, doc, state));
    }
    out
}

fn line_for<'a>(raw: &'a str, line_no: usize, doc: &'a Doc, state: &ReaderState) -> Line<'a> {
    // Heading?
    for h in &doc.headings {
        if h.line == line_no {
            return Line::from(vec![
                Span::raw(" "),
                Span::styled("▎ ", Style::default().fg(theme::ROSE)),
                Span::styled(
                    h.text.clone(),
                    Style::default()
                        .fg(theme::ROSE)
                        .add_modifier(Modifier::BOLD),
                ),
            ]);
        }
    }
    // Task?
    if let Some((task_i, task)) = doc.tasks.iter().enumerate().find(|(_, t)| t.line == line_no) {
        let focused = state.focused_task == Some(task_i);
        let marker = if focused { "▸ " } else { "  " };
        let box_str = if task.checked { "[x]" } else { "[ ]" };
        let box_style = if task.checked {
            Style::default().fg(theme::CHECK)
        } else {
            Style::default().fg(theme::DIM)
        };
        let text_style = if task.checked {
            Style::default().fg(theme::DIM)
        } else {
            Style::default().fg(theme::TEXT)
        };
        let mut line = Line::from(vec![
            Span::raw("   "),
            Span::raw(marker.to_string()),
            Span::styled(box_str, box_style),
            Span::raw(" "),
            Span::styled(task.text.clone(), text_style),
        ]);
        if focused {
            line = line.style(Style::default().bg(theme::BG_FOCUS));
        }
        return line;
    }
    // Plain prose: indent 4 cols from gutter.
    if raw.trim().is_empty() {
        Line::from("")
    } else {
        Line::from(Span::raw(format!("     {raw}")))
    }
}

    let Some(doc) = app.current_doc().cloned() else {
        app.view = View::Index;
        return;

use ratatui::Frame;

#[derive(Debug, Default, Clone)]
pub struct OutlineState {
    pub focus: usize,
}

pub fn render(f: &mut Frame, area: Rect, doc: &Doc, state: &OutlineState) {
    let modal = centered_rect(70, 70, area);
    f.render_widget(Clear, modal);

    let block = Block::default()
        .borders(Borders::ALL)
        .border_style(Style::default().fg(theme::ROSE))
        .title(Span::styled(
            " outline ",
            Style::default().fg(theme::TITLE).add_modifier(theme::TITLE_MOD),
        ));
    let inner = block.inner(modal);
    f.render_widget(block, modal);

    let mut lines = Vec::new();
    for (i, h) in doc.headings.iter().enumerate() {
        let indent = " ".repeat(((h.level.saturating_sub(1)) as usize) * 2);
        let marker = if i == state.focus { "▸ " } else { "  " };
        let style = if i == state.focus {
            Style::default()
                .fg(theme::TEXT)
                .bg(theme::BG_FOCUS)
                .add_modifier(Modifier::BOLD)
        } else {
            Style::default().fg(theme::TEXT)
        };
        lines.push(Line::from(Span::styled(
            format!(" {marker}{indent}{}", h.text),
            style,
        )));
    }
    if lines.is_empty() {
        lines.push(Line::from(Span::styled(
            " (no headings)",
            Style::default().fg(theme::DIM),
        )));
    }

    let para = Paragraph::new(lines).style(Style::default().bg(theme::BG_BASE));
    f.render_widget(para, inner);
}

//! Surgical checkbox toggle with backup file + atomic rename.
use anyhow::{anyhow, Result};
use std::path::{Path, PathBuf};

/// Toggle the checkbox on `line_no` (0-indexed) in the file at `path`.
/// Writes a backup copy to `backups_dir` first; on success returns the new
/// `checked` value.
pub fn toggle_checkbox(path: &Path, line_no: usize, backups_dir: &Path) -> Result<bool> {
    let src = std::fs::read_to_string(path)?;
    let mut lines: Vec<String> = src.split_inclusive('\n').map(|s| s.to_string()).collect();
    let target = lines
        .get(line_no)
        .ok_or_else(|| anyhow!("line {line_no} out of range in {}", path.display()))?
        .clone();

    let new_state = if let Some(pos) = target.find("- [ ]") {
        let mut rep = target.clone();
        rep.replace_range(pos..pos + 5, "- [x]");
        lines[line_no] = rep;
        true
    } else if let Some(pos) = target.find("- [x]").or_else(|| target.find("- [X]")) {
        let mut rep = target.clone();
        rep.replace_range(pos..pos + 5, "- [ ]");
        lines[line_no] = rep;
        false
    } else {
        return Err(anyhow!(
            "line {line_no} is not a checkbox line: {:?}",
            target.trim_end()
        ));
    };

    // Backup.
    write_backup(path, backups_dir)?;

    // Atomic rename.
    let tmp = unique_tmp_path(path);
    std::fs::write(&tmp, lines.join(""))?;
    if let Err(e) = std::fs::rename(&tmp, path) {
        // Cross-filesystem? Fall back to copy + delete.
        std::fs::copy(&tmp, path)?;
        std::fs::remove_file(&tmp)?;
        eprintln!("atomic rename failed; copied instead: {e}");
    }

    Ok(new_state)
}

fn unique_tmp_path(target: &Path) -> PathBuf {
    let mut tmp = target.to_path_buf();
    let stem = target
        .file_name()
        .and_then(|n| n.to_str())
        .unwrap_or("atlas-edit");
    tmp.set_file_name(format!(".{stem}.atlas-tmp.{}", std::process::id()));
    tmp
}

fn write_backup(src_path: &Path, backups_dir: &Path) -> Result<()> {
    std::fs::create_dir_all(backups_dir)?;
    let stem = src_path
        .file_name()
        .and_then(|n| n.to_str())
        .unwrap_or("atlas-edit");
    let ts = std::time::SystemTime::now()
        .duration_since(std::time::UNIX_EPOCH)
        .map(|d| d.as_secs())
        .unwrap_or(0);
    let backup = backups_dir.join(format!("{stem}.{ts}.md"));
    std::fs::copy(src_path, &backup)?;
    prune_backups(backups_dir, stem, 20)?;
    Ok(())
}

fn prune_backups(dir: &Path, stem: &str, max: usize) -> Result<()> {
    let mut matches: Vec<_> = std::fs::read_dir(dir)?
        .filter_map(|e| e.ok())
        .filter(|e| {
            e.file_name()
                .to_string_lossy()
                .starts_with(&format!("{stem}."))
        })
        .collect();
    if matches.len() <= max {
        return Ok(());
    }
    matches.sort_by_key(|e| e.metadata().and_then(|m| m.modified()).ok());
    for e in matches.iter().take(matches.len() - max) {
        let _ = std::fs::remove_file(e.path());
    }
    Ok(())
}

//! OSC 52 clipboard write.  Caps payload at 4 KiB raw bytes pre-base64
//! (Termux pty buffer limit; Blink times out on more).
use base64::engine::general_purpose::STANDARD;
use base64::Engine;
use std::io::Write;

pub const MAX_BYTES: usize = 4096;

/// Returns true if the payload was truncated.
pub fn copy(data: &str) -> bool {
    let (payload, truncated) = if data.as_bytes().len() > MAX_BYTES {
        // Truncate at a char boundary.
        let mut end = MAX_BYTES;
        while !data.is_char_boundary(end) && end > 0 {
            end -= 1;
        }
        (&data[..end], true)
    } else {
        (data, false)
    };
    let encoded = STANDARD.encode(payload.as_bytes());
    let mut out = std::io::stdout().lock();
    let _ = write!(out, "\x1b]52;c;{encoded}\x07");
    let _ = out.flush();
    truncated
}

    let src = format!(

fn buf_to_string(buf: &ratatui::buffer::Buffer) -> String {
    let mut out = String::new();

# atlas

Flagship TUI reader for plan-shaped markdown. Library of plans, roadmaps, and tasks rendered with care.

See `docs/superpowers/specs/2026-05-27-atlas-design.md` for the design.

## Build + install

    ./install.sh

Installs `atlas` to both `~/.cargo/bin` and `~/.local/bin` (suite convention; whichever is first on PATH wins).

## Config

`~/.config/atlas/library.toml`:

    roots = [
      "~/projects/*/docs/superpowers/plans/*.md",
      "~/projects/*/ROADMAP.md",
      "~/vaults/sops/plans/*.md",
    ]

Atlas only surfaces docs that have a YAML frontmatter block declaring `type: plan|roadmap|task`. Add this to the top of any doc you want indexed:

    ---
    type: plan
    title: cal-design
    status: active
    ---

Optional fields:

- `title` (defaults to file stem)
- `status`: active (default) / shipped / dropped
- `progress`: integer 0-100, only used if the doc has no checkbox tasks

## Keys

Press `?` inside atlas for the full keymap.

Index: `j/k` move, `enter` open, `/` filter, `e` open in `$EDITOR`, `y` copy path, `q` quit.

Reader: `j/k` scroll / move task focus, `J/K` jump section, `space` toggle checkbox, `o` outline, `/` search, `h` back, `q` quit.

## State

- `~/.config/atlas/library.toml` configures discovery.
- `~/.config/atlas/state.json` remembers your last doc and scroll position. `atlas --no-resume` ignores it on startup.
- `~/.cache/atlas/backups/<basename>.<unix_ts>.md` keeps the last 20 backups per file before every checkbox toggle.
